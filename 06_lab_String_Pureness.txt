num = int(input())
for word in range(num):
    word1 = input()
    is_not_pure_string = True
    for i in word1:
        if i in ('_','.',','):
            is_not_pure_string = False

            print(f"{word1} is not pure!")
            break
    if is_not_pure_string:
        print(f"{word1} is pure.")
