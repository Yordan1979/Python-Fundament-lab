num_receive_water = int(input())
counter_water_liters = 0
for water in range(num_receive_water):
    receive_liters_water = int(input())
    counter_water_liters += receive_liters_water
    if counter_water_liters > 255:
        print("Insufficient capacity!")
        counter_water_liters -= receive_liters_water
print(counter_water_liters)
