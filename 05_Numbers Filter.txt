n_numbers = int(input())
lst_numbers = []
lst_filtered_numbers = []
command = ''
for n in range(n_numbers):
    numbers = int(input())
    lst_numbers.append(numbers)
command = input()
if command == 'even':
    for numbers in lst_numbers:
        if numbers % 2 == 0 or numbers == 0:
            lst_filtered_numbers.append(numbers)
elif command == 'odd':
    for numbers in lst_numbers:
        if numbers % 2 != 0:
            lst_filtered_numbers.append(numbers)
elif command == 'negative':
    for numbers in lst_numbers:
        if numbers < 0:
            lst_filtered_numbers.append(numbers)
elif command == 'positive':
    for numbers in lst_numbers:
        if numbers >= 0:
            lst_filtered_numbers.append(numbers)

print(lst_filtered_numbers)
